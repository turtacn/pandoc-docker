构建镜像

# 使用 Docker Compose 构建 (推荐)
docker-compose build

# 或者使用原生 Docker 命令
docker build -t jdcloudiaas/turta:pandoc .

运行容器并转换文档

# 启动交互式 Shell
docker-compose run --rm jdcloudiaas/turta:pandoc /bin/bash

# 在容器内执行转换命令
md2pdf.sh test/sample.md          # 基本的 PDF 转换
md2pdf.sh test/sample.md --toc    # 生成带目录的 PDF
md2docx.sh test/sample.md         # 转换成 DOCX
md2pptx.sh presentation.md        # 转换成 PPTX
